decision = shield.route("0x1234...abcd")

if decision.recommendation == "instant_settle":
    settle_payment(amount)
elif decision.recommendation == "escrow":
    create_escrow(amount, timeout="24h")
else:

from ahm_shield import AHMShield

shield = AHMShield(api_key="ahm_live_...")

@shield.guard(min_grade="C")
def process_payment(sender: str, amount: float):
    # Only runs if sender's AHS grade >= C
    # Raises AHMRejectError for D/E/F grades
    # Raises AHMEscrowFlag for escrow recommendations
    transfer(sender, amount)
    return {"status": "settled"}
